// --- [LicenseManager.cs] ---
using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;
using Microsoft.Win32;
using Autodesk.Revit.UI;
using System.Management;

namespace TilePlanner.Security
{
    public static class LicenseManager
    {
        private const string RegistryPath = @"Software\AntiGravity\TilePlanner";
        private const string LicenseKey = "LicenseToken";
        private const string TrialDays = "90";
        private static readonly byte[] Salt = Encoding.UTF8.GetBytes("AntiGravity_Salt_2026");

        public static bool Validate()
        {
            try
            {
                string machineId = GetMachineId();
                RegistryKey key = Registry.CurrentUser.OpenSubKey(RegistryPath, true);
                if (key == null) return InitializeLicense(machineId);

                string encryptedToken = key.GetValue(LicenseKey) as string;
                if (string.IsNullOrEmpty(encryptedToken)) return InitializeLicense(machineId);

                string decryptedInfo = Decrypt(encryptedToken, machineId);
                string[] parts = decryptedInfo.Split('|');
                if (parts.Length != 3) return false;

                string registeredMachineId = parts[0];
                DateTime installDate = DateTime.Parse(parts[1]);
                DateTime lastRunDate = DateTime.Parse(parts[2]);

                if (registeredMachineId != machineId)
                {
                    TaskDialog.Show("授權錯誤", "檢測到硬體變更。請聯繫管理員。");
                    return false;
                }

                DateTime now = DateTime.Now;
                if (now < lastRunDate)
                {
                    TaskDialog.Show("安全鎖定", "檢測到系統時間異常（時鐘回溯）。");
                    return false;
                }

                int remainingDays = (int)(installDate.AddDays(double.Parse(TrialDays)) - now).TotalDays;
                if (remainingDays < 0)
                {
                    TaskDialog.Show("授權已過期", "您的 90 天試用期已結束。");
                    return false;
                }

                UpdateLastRun(key, machineId, installDate, now);
                if (remainingDays <= 7)
                    TaskDialog.Show("授權即將到期", $"您的試用授權還剩餘 {remainingDays} 天。");

                return true;
            }
            catch (Exception ex)
            {
                TaskDialog.Show("授權驗證失敗", ex.Message);
                return false;
            }
        }

        private static bool InitializeLicense(string machineId)
        {
            RegistryKey key = Registry.CurrentUser.CreateSubKey(RegistryPath);
            DateTime now = DateTime.Now;
            string token = Encrypt($"{machineId}|{now:O}|{now:O}", machineId);
            key.SetValue(LicenseKey, token);
            TaskDialog.Show("歡迎使用", "TilePlanner 已透過專業安裝程式啟動！試用期為 90 天。");
            return true;
        }

        private static void UpdateLastRun(RegistryKey key, string machineId, DateTime installDate, DateTime now)
        {
            string token = Encrypt($"{machineId}|{installDate:O}|{now:O}", machineId);
            key.SetValue(LicenseKey, token);
        }

        private static string GetMachineId()
        {
            string cpuId = "";
            try {
                ManagementObjectSearcher searcher = new ManagementObjectSearcher("Select ProcessorId From Win32_Processor");
                foreach (ManagementObject mo in searcher.Get()) cpuId = mo["ProcessorId"].ToString();
            } catch { cpuId = "UNKNOWN"; }
            return cpuId;
        }

        private static string Encrypt(string text, string key)
        {
            using (Aes aes = Aes.Create())
            {
                var pdb = new Rfc2898DeriveBytes(key, Salt, 1000);
                aes.Key = pdb.GetBytes(32);
                aes.IV = pdb.GetBytes(16);
                using (var ms = new MemoryStream())
                {
                    using (var cs = new CryptoStream(ms, aes.CreateEncryptor(), CryptoStreamMode.Write))
                    {
                        byte[] data = Encoding.UTF8.GetBytes(text);
                        cs.Write(data, 0, data.Length);
                    }
                    return Convert.ToBase64String(ms.ToArray());
                }
            }
        }

        private static string Decrypt(string cipherText, string key)
        {
            using (Aes aes = Aes.Create())
            {
                var pdb = new Rfc2898DeriveBytes(key, Salt, 1000);
                aes.Key = pdb.GetBytes(32);
                aes.IV = pdb.GetBytes(16);
                using (var ms = new MemoryStream())
                {
                    using (var cs = new CryptoStream(ms, aes.CreateDecryptor(), CryptoStreamMode.Write))
                    {
                        byte[] data = Convert.FromBase64String(cipherText);
                        cs.Write(data, 0, data.Length);
                    }
                    return Encoding.UTF8.GetString(ms.ToArray());
                }
            }
        }
    }
}
